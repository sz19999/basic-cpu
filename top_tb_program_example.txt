~ basic CPU program ~

mvi R0, $6	# R0 <= 6
mvi R1, $1	# R1 <= 1
add R1, R0      # R1 <= R1 + R0 = 1 + 6 = 7 
mv  R2, R1      # R2 <= R1 = 6
mvi R3, $15     # R3 <= 15     
sub R3, R0      # R3 <= R3 - R0 = 15 - 6 = 9
mv  R4, R0      # R4 <= R0 = 6
sub R4, R1      # R4 <= R4 - R1 = 6 - 7 = -1
mvi R5, $5	# R5 <= 5
mvi R6, $3	# R6 <= 3
add R6, R5	# R6 <= R6 + R5 = 3 + 5 = 8
nop		# do nothing
mv R7, R6	# R7 <= R6 = 8
